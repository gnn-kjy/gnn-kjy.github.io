3、有效数字【中等】

判断一串字符串是否为有效数字，enum枚举自定义数据类型，unordered_map实现状态转换机，字符只能算函数实现分类问题

class Solution {
public:
    enum CharType {
        CT_SPACE,
        CT_OP,
        CT_INT,
        CT_POINT,
        CT_EXP,
        CT_ILLEGAL
    };
    CharType toCharType(char c) {
        if(c==' ')  return CT_SPACE;
        else if(c=='+'||c=='-') return CT_OP;
        else if(c>='0'&&c<='9') return CT_INT;
        else if(c=='.') return CT_POINT;
        else if(c=='e'||c=='E') return CT_EXP;
        else return CT_ILLEGAL;
    }
    enum State {
        S_INIT,
        S_OP,
        S_INT,
        S_INT_POINT,//点前是整数
        S_WITHOUT_INT_POINT,//点前无整数
        S_FRACTION,//小数
        S_EXP,
        S_EXP_OP,//其实不细分exp的操作符和后面整数应该也是没问题的，就是最终状态判断的时候有区别，细分更踏实点
        S_EXP_INT,
        S_END
    };
    bool validNumber(string s) {
        unordered_map<State, unordered_map<CharType, State>> StateList {
            {
                S_INIT, {
                    {CT_SPACE, S_INIT},
                    {CT_OP, S_OP},
                    {CT_INT, S_INT},
                    {CT_POINT, S_WITHOUT_INT_POINT}
                }
            },
            {
                S_OP, {
                    {CT_INT, S_INT},
                    {CT_POINT, S_WITHOUT_INT_POINT}
                }
            },
            {
                S_INT, {
                    {CT_INT, S_INT},
                    {CT_POINT, S_INT_POINT},
                    {CT_EXP, S_EXP},
                    {CT_SPACE, S_END}
                }
            },
            {
                S_INT_POINT, {
                    {CT_INT, S_FRACTION},
                    {CT_EXP, S_EXP},
                    {CT_SPACE, S_END}
                }
            },
            {
                S_WITHOUT_INT_POINT, {
                    {CT_INT, S_FRACTION}
                }
            },
            {
                S_FRACTION, {
                    {CT_INT, S_FRACTION},
                    {CT_EXP, S_EXP},
                    {CT_SPACE, S_END}
                }
            },
            {
                S_EXP, {
                    {CT_OP, S_EXP_OP},
                    {CT_INT, S_EXP_INT}
                }
            },
            {
                S_EXP_OP, {
                    {CT_INT, S_EXP_INT}
                }
            },
            {
                S_EXP_INT, {
                    {CT_INT, S_EXP_INT},
                    {CT_SPACE, S_END}
                }
            },
            {
                S_END, {
                    {CT_SPACE, S_END}
                }
            }
        };

        State st=S_INIT;
        for(int i=0;i<s.length();i++) {
            if(toCharType(s[i])==CT_ILLEGAL)    return false;//非法字符
            // if(StateList[st].find(s[i])==StateList[st].end())   return false;错在查找的应该是自定义的字符类型
            if(StateList[st].find(toCharType(s[i]))==StateList[st].end())   return false;//不在状态转换表内
            // st = StateList[st][s[i]]错在没有分号和s[i]
            st = StateList[st][toCharType(s[i])];
        }
        if(st==S_END||st==S_EXP_INT||st==S_INT||st==S_FRACTION||st==S_INT_POINT)    return true;
        return false;
    }
};
4、把字符串转换成整数（atoi）【中等】

string到int类型转换，条件循环考虑全面，字符减法循环10次幂实现字符转整数

class Solution {
public:
    int myAtoi(string str) {
        string s="";//定义或者使用空字符串不可用单引号
        int i=0, len=str.length(), flag=0;
        while(str[i]==' '&&i<len) {//跳过前导空格
            i++;
        }
        if(i<len&&(str[i]=='+'||str[i]=='-')) {
            if(str[i]=='-') {
                flag=1;//0正1负
            }
            i++;
        }
        while(i<len&&str[i]=='0') {//跳过前导0
            i++;
        }
        while(i<len&&(str[i]>='0'&&str[i]<='9')) {
            s+=str[i];
            i++;
        }

        int ans=0;
        if(s=="")   return ans;//定义或者使用空字符串不可用单引号
        int len1=s.length();
        if(flag==0) {
            for(int i=0;i<s.length();i++) {
                if((ans+(s[i]-'0')*pow(10,len1-1))>=pow(2,31)-1)    return pow(2,31)-1;
                ans+=(s[i]-'0')*pow(10,--len1);
            }
            return ans;
        } else {
            for(int i=0;i<s.length();i++) {
                if((ans+(s[i]-'0')*pow(10,len1-1))>=pow(2,31))    return 0-pow(2,31);
                ans+=(s[i]-'0')*pow(10,--len1);
            }
            return 0-ans;
        }
    }
};
【小结】

  • ‘encryption’—加密

  • ‘size()’—vector/string；’length’—string

  • ‘substr(pos,len)’—string—起始位置，长度

  • string重载了运算符’+’

  • enum枚举大括号外为’;’，括号内’,’分隔

  • ‘函数名’首字母小写，其余分词首字母均大写；’数据类型’每个分次首字母均大写

  • 定义或者使用空string字符串不可用单引号’’，一定要用双引号””

  • atoi—ascii to integer—爱死尅 特 英题杰

  • unordered_map—无序map容器—使用<>{}嵌套定义—大括号外为;—内部元素分隔为,—使用时只能是<>内部写的数据类型

  • 返回负整数不可直接前面加负号—使用0-x实现

  • 字符串转整数根本思想就是循环(‘x’-‘0’)*pow(10,y)—只不过具体题目幂底数或字符串复杂度有差异

/**

class Solution {
public:

        vector<int> ans;

        }

};

3、复杂链表的复制【中等】

随机链表复制，利用无序map容器的key-value映射节点-节点，重新复制构造每个节点（值val），将节点串联起来（两个指针next、random重新串起来）。关键在于分清[]内调用指针（原节点的指针）和[]外调用指针（新建节点的指针）的含义。

/*
// Definition for a Node.
class Node {
public:
    int val;
    Node* next;
    Node* random;
    
    Node(int _val) {
        val = _val;
        next = NULL;
        random = NULL;
    }
};
*/
class Solution {
public:
    Node* copyRandomList(Node* head) {
        Node* ptr = head;
        unordered_map<Node*, Node*> tmp;
        while(ptr) {
            tmp[ptr] = new Node(ptr->val);
            ptr=ptr->next;
        }
        ptr=head;
        while(ptr) {
            tmp[ptr]->next=tmp[ptr->next];
            tmp[ptr]->random=tmp[ptr->random];
            ptr=ptr->next;
        }
        return tmp[head];
    }
};
【小结】
  • 看准函数返回值类型和参数类型
  • 指针不等价于链表，指针取内容的*不适用于链表，链表用箭头？
  • vector对于数组的插入操作—insert(x,y)任意位置插，x是指针，y是要插内容—push_back(y)尾插—begin()和end()返回头尾指针
  • 链表倒序—中间容器—多个头指针—本质是原链表指向不变，只修改val
  • 并行定义链表节点每个变量前都要加*
  • 无序map容器可用来复制复杂链表

链表—结构体指针(单)/Node指针(多)—对链表的操作分为两大步：先处理val(一般有顺序要求)，再串联指针。

class Solution {
public:
    ListNode* deleteNode(ListNode* head, int val) {
        if(head->val==val)  return head->next;

        ListNode* ans=head;
        while(head) {
            ListNode* tmp=head->next;
            if(tmp->val==val) {
                head->next=tmp->next;
                return ans;
            }
            head=head->next;
        }
        return ans;
    }
};
2、倒第n节点链表

不是返回单个节点，是倒第cnt个节点和其之后的一整条，统计链表长度来实现链表。(本来想构造逆序数组，使用cnt作为数组编号查找，但是链表节点的val可能重复)

/**
 * 第一时间想到的就是先统计长度，然后遍历
*/
class Solution {
public:
    ListNode* trainingPlan(ListNode* head, int cnt) {
        int len=0, i=0;
        ListNode* ans=head;
        while(head) {
            len++;
            head=head->next;
        }
        while(ans) {
            i++;
            if(i==(len-cnt+1)) {
                return ans;
            }
            ans=ans->next;
        }
        return ans;
    }
};
/**
 * 双指针：但是这章专题是双指针，利用前后指针间距实现倒第几个节点的查找
*/
class Solution {
public:
    ListNode* trainingPlan(ListNode* head, int cnt) {
        ListNode* former=head, *latter=head;
        while(cnt--) {
            former=former->next;//间隔控制
        }
        while(former) {
            latter=latter->next;
            former=former->next;
        }
        return latter;//注意上边循环终止刚好是latter返回，不是latter->next
    }
};
3、升序拼接链表

先遍历拼接，然后构造数组并排序，重新串联链表，返回链表。(需要几个重新指向头结点的指针等用的时候再回头定义。)还有就是注意l1为空时比较特殊。还有拼接之后l1就变长了，注意拼完立刻中断。

class Solution {
public:
    ListNode* trainningPlan(ListNode* l1, ListNode* l2) {
        ListNode* t1=l1, * t2=l1, * t3=l1;
        if(!l1) {
            t1=t2=t3=l2;
        }
        while(l1) {
            if(l1->next==nullptr) {//这里双等号
                l1->next=l2;//这里接上之后得终止，因为l1延长了，不然死循环
                break;
            }
            l1=l1->next;
        }
        vector<int> a;
        while(t1) {
            a.push_back(t1->val);
            t1=t1->next;
        }
        sort(a.begin(), a.end());
        for(int i=0;i<a.size();i++) {
            t2->val=a[i];
            t2=t2->next;
        }
        return t3;
    }
};
4、链表交叉节点

一个锻炼项目只对应一个唯一的编号，所以这题只用找出相同的值就行了。而且从后往前找更方便。所以倒插，以最短为标准，分为不相交、交一部分、完全重叠三种情况。另外考虑链表为空的情况即可。

class Solution {
public:
    ListNode *getIntersectionNode(ListNode *headA, ListNode *headB) {
        vector<int> tmp1, tmp2;
        if(!headA||!headB)  return NULL;
        while(headA) {
            tmp1.insert(tmp1.begin(), headA->val);
            headA=headA->next;
        }
        while(headB) {
            tmp2.insert(tmp2.begin(), headB->val);
            headB=headB->next;
        }
        if(tmp1[0]!=tmp2[0])    return NULL;
        int i=0;
        int len=(tmp1.size()<=tmp2.size()?tmp1.size():tmp2.size());
        for(i;i<len;i++) {
            if(tmp1[i]!=tmp2[i]) {
                ListNode *ans=new ListNode(tmp1[i-1]);
                return ans;
            }
            if(i==len-1) {
                ListNode *ans=new ListNode(tmp1[i]);
                return ans;
            }
        }
        return NULL;
    }
};

7、字符串中的单词反转

借助一个暂时字符串变量和一个字符串容器进行中转，最后拼接。(注意前导零和尾零的处理比较特殊，也插，只不过元素为空，还是占用容器空间)

class Solution {
public:
    string reverseMessage(string message) {
        vector<string> s;
        int n=0;
        while(n<message.size()) {
            while(n<message.size()&&message[n]==' ') {
                n++;
            }
            string tmp="";
            while(n<message.size()&&message[n]!=' ') {
                tmp+=message[n];
                n++;
            }
            s.insert(s.begin(),tmp);
        }
        string ans="";
        for(int i=0;i<s.size();i++) {
            if((i<s.size()-1)&&s[i]!="") {
                ans+=s[i];
                ans+=" ";
            } else {
                ans+=s[i];
            }
        }
        return ans;
    }
};

但是时间复杂度是n²，而且这题在双指针这里，前指针往前移，后指针跟着截取，用substr截取。最后再去掉最后的空格。从后往前遍历保证了顺序是反的要求。

class Solution {
public:
    string reverseMessage(string message) {
        int len=message.size();
        string ans="";
        for(int i=len-1;i>=0;i--) {
            if(message[i]!=' ') {
                int right=i;
                while(i>=0&&message[i]!=' ') {
                    i--;
                }
                ans+=message.substr(i+1,right-i)+" ";//这样最后肯定会多一个空格
            }
        }
        return ans.substr(0,ans.size()-1);//截取一下
    }
};

class CQueue {
public:
    vector<int> a;
    CQueue() {
        a.push_back(0);//0打底，否则迭代器越界，也正好满足下面能够循环起来并最后返回-1
    }
    
    void appendTail(int value) {
        a.push_back(value);
    }
    
    int deleteHead() {
        if(a.begin()!=a.end()-1) {//a.end()-1为最后一个元素
            a.erase(a.begin());
            return a[0];//先删前面被取过的，再取，能循环起来。最后取完后数组里还剩一个被取过的，直接返回-1
        } else {
            return -1;
        }
    }
};

 * Your CQueue object will be instantiated and called as such:
 * CQueue* obj = new CQueue();
 * obj->appendTail(value);
 * int param_2 = obj->deleteHead();
 */
2、最小栈【简单】

借助数组处理栈，然后就是简单的入栈出栈，依次比较返回最小值

class MinStack {
public:
    /** initialize your data structure here. */
    vector<int> a;
    MinStack() {}
    
    void push(int x) {
        a.push_back(x);
    }
    
    void pop() {
        a.erase(a.end()-1);
    }
    
    int top() {
        int len=a.size();
        return a[len-1];
    }
    
    int getMin() {
        int n=INT_MAX, m=a.size();
        while(m--) {
            if(a[m]<n) {
                n=a[m];
            }
        }
        return n;
    }
};

3、望远镜最高海拔【困难】

滑动窗口的最大值，跟栈倒是没太大关系，借助一个局部最大值变量和双指针操作数组；注意数组空的情况。

class Solution {
public:
    vector<int> maxAltitude(vector<int>& heights, int limit) {
        vector<int> ans;
        if(heights.size()==0) {
            return ans;
        }
        for(int former=0;former<=heights.size()-limit;former++) {
            int latter=former;
            int max=INT_MIN;
            while(latter-former<=limit-1) {
                if(heights[latter]>max) {
                    max=heights[latter];
                }
                latter++;
            }
            ans.push_back(max);
        }
        return ans;
    }
};

五、模拟

1、螺旋遍历二维数组【简单】

总感觉这题是简单不合理；双重循环，外层一循环内层四循环；四边界约束缩减遍历(rb=row_begin；re=row_end；cb=column_begin；ce=column_end)

class Solution {
public:
    vector<int> spiralArray(vector<vector<int>>& array) {
        vector<int> ans;
        if(array.size()==0) return ans;
        int rb=0, cb=0, re=array.size()-1, ce=array[0].size()-1;//下标，别忘记减一
        while(1) {
            for(int i=cb; i<=ce; i++) {
                ans.push_back(array[rb][i]);//→
            }
            if(++rb>re) break;
            for(int i=rb; i<=re; i++) {
                ans.push_back(array[i][ce]);//↓
            }
            if(--ce<cb) break;
            for(int i=ce; i>=cb; i--) {
                ans.push_back(array[re][i]);//←
            }
            if(--re<rb) break;
            for(int i=re; i>=rb; i--) {
                ans.push_back(array[i][cb]);//↑
            }
            if(++cb>ce) break;            
        }
        return ans;
    }
};
2、验证图书取出顺序【中等】

类似四则运算或前缀后缀表达式转换，符号匹配；借助栈解决；先压入，再判断，相等则出栈，否则继续压入；前插，否则没办法取最后一个元素

class Solution {
public:
    bool validateBookSequences(vector<int>& putIn, vector<int>& takeOut) {
        if(takeOut.size()==0)   return true;
        if(putIn.size()==0) return false;
        vector<int> stack;
        int j=0;
        for(int i=0; i<putIn.size(); i++) {
            stack.insert(stack.begin(), putIn[i]);
            while(!stack.empty()&&takeOut[j]==stack[0]) {//是and不是or
                stack.erase(stack.begin());
                j++;
            }
        }
        return stack.empty();
    }
};
【小结】
  • 螺旋遍历二维数组—四边界约束遍历
  • 数组—注意下标和大小
  • vector容器的erase和remove—remove将等于value的元素放到vector的尾部，不减少vector的size—erase删除一个/段元素，删除后原迭代器的位置不变
  • vector容器为空与否—empty()函数—size()函数

6、寻找第一个单一字母【简单】

双重循环暴力解法，统计每个字母出现的次数

class Solution {
public:
    char dismantlingAction(string arr) {
        int count=0;
        for(int i=0;i<arr.length();i++) {
            for(int j=0;j<arr.length();j++) {
                if(arr[i]==arr[j]) {
                    count++;
                }
                if(count==2)    break;
            }
            if(count==1)    return arr[i];
            count=0;
        }
        return ' ';
    }
};
【小结】
  • 寻找重复元素—排序
  • 非严格递增—不排除相等情况的递增
  • 在一些循环类问题中，自以为的可以省时省内存的提前终止的判断语句都是冗余的
